% \curvedLink[thickness]{from}{to}{label}{control point or points}
\providecommand{\curvedLink}[5][]{\draw[->,shorten >=2pt,line width=1.3pt,#1] (#2) .. controls #5 .. node[below, sloped] {#4} (#3);}

\providecommand{\straightLink}[4][]{\draw[line width=1pt,->,shorten >=2pt,#1] (#2) -- (#3) node[pos=0.85,above,sloped,#1]{#4};}

\providecommand{\inhibStraightLink}[4][line width=1pt]{\draw[ ->,style=densely dashed,shorten >=2pt,#1] (#2) -- (#3) node[pos=0.7,above,sloped]{#4};}
